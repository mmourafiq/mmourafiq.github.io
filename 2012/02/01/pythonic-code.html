Make your code pythonic

Some python best practices to follow.

Enemurate

 1 #  example of bad code
 2 for indice in xrange(0, len(l)):
 3     print "l[%d] = %r" % (indice, l[indice])
 4 
 5 # ==>
 6 #  the best way to perform such operation (get indices and corresponding values) is by using enumerate
 7 
 8 for indice, value in enumerate(l):
 9     print "l[%d] = %r" % (indice, value)
10 #  since enumerate returns a list of tuples :
11 for i in enumerate(l):
12     print "l[%d] = %r" % i

Generators

 1 #  simple range
 2 def my_range(x, y=0):
 3     if y < x: # inverse if y < x
 4         x, y = y, x
 5 
 6     yield x # return the first x
 7     while x < y:
 8         x += 1
 9         yield x
10 
11 #  simple enumerate
12 def enumerate(i):
13     for i in xrange(0, len(i)):
14         yield i, i[i]
15 
16 #  simple xrange
17 def xrange(start, stop=None, step=1):
18     if stop is None:
19         start, stop = 0, start
20     while step > 0 and start < stop or step < 0 and start > stop:
21         yield start
22         start += step

Coroutines

 1 def cpt(x):
 2     n = 0
 3     while n <= x:
 4         v = (yield n)
 5         if v is not None:
 6             n = v
 7         else:
 8             n += 1
 9 
10 gen = cpt(20)
11 for i in gen:
12     print i
13     # we want to go from 15 to 18
14     if i == 15:
15         gen.send(18)

Zip

 1 #  suppose we want to iterate over 2 list l1 & l2
 2 #  the easiest way to do is :
 3 for i in xrange(min(len(l1), len(l2))):
 4     v1, v2 = l1[i], l2[i]
 5     # do something with v1, v2
 6 
 7 #  the easiest way to do so:
 8 for v1, v2 in zip(l1, l2):
 9     # do something with v1, v2
10 
11 #  zip definition:
12 #  zip(...)
13     # zip(seq1 [, seq2 [...]]) -> [(seq1[0], seq2[0] ...), (...)]
14     # Return a list of tuples, where each tuple contains the i-th element
15     # from each of the argument sequences. The returned list is truncated
16     # in length to the length of the shortest argument sequence.
17 
18 #  e.g
19 print zip(['a', 'b', 'c'], ['d', 'e', 'f']) # [('a', 'd'), ('b', 'e'), ('c', 'f')]
20 
21 #  python doesn't provide an unzip function but we can easily code one:
22 unzip = lambda l: [list(li) for li in zip(*l)]
23 unzip([(1, 2), (3, 4), (5, 6)]) # [[1, 3, 5], [2, 4, 6]]
